The below is Movie class structure: 

                class Movie{
                    title:string;
                    studio:string;
                    rating:string;
                
                    constructor(movienm:string, studio:string,rating:string='PG')
                    {
                        this.title=movienm;
                        this.studio=studio;
                        this.rating=rating;
                    }
                
                    getPG(moviesarr:Array):Array{
                        let pgMovies:Array=[];
                        moviesarr.forEach(movie=>{
                            if(movie.rating==='PG')
                                pgMovies.push(movie);
                        })
                        return pgMovies;
                    }
                }
